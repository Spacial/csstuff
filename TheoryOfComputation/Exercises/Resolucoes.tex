\begin{center}

    \node[accepting,state]         (B) [right of=A] 	{$q_1$};
    \node[state,accepting]         (D) [right of=B] 	{$q_2$};
    \node[state]         (C) [below of=A] 	{$q_3$};
    \node[state]         (E) [below of=B]       {$q_4$};
    \node[state]         (F) [below of=D]       {$q_5$};
    \node[state]         (G) [right of=D]       {$q_6$};

	      edge [bend left]  node {0} (C)
	  (B) edge 		node {1} (D)
	      edge [bend left]  node {0} (E)
	  (C) edge              node {1} (E)
	      edge [bend left]  node {0} (A)
	  (D) edge 		node {1} (G)
	      edge [bend left]  node {0} (F)
	  (E) edge              node {1} (F)
	      edge [bend left]  node {0} (B)
	  (F) edge 		node {1} (G)
	      edge [bend left]  node {0} (D)	  
	  (G) edge [loop right]	node {1,0} (G);
	  
  \end{tikzpicture}\end{center}
